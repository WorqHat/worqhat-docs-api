# Delete Data

Deletes records from the specified table that match the given conditions. Organization ID is automatically included in the conditions for multi-tenant security.

```endpoint
POST /db/delete
```

## Request Body

<ParamField body="table" type="string" required>
  Table name to delete records from
</ParamField>

<ParamField body="conditions" type="object" required>
  Conditions to match records for deletion
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the delete operation was successful
</ResponseField>

<ResponseField name="deletedCount" type="integer">
  Number of records deleted
</ResponseField>

<ResponseField name="message" type="string">
  Success message
</ResponseField>

<RequestExample>
```bash cURL
curl -X POST "https://api.worqhat.com/db/delete" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -d '{
    "table": "users",
    "conditions": {
      "status": "inactive",
      "last_login_before": "2024-01-01T00:00:00Z"
    }
  }'
```

```javascript Node.js
import Worqhat from 'worqhat';

const client = new Worqhat({
  apiKey: process.env.WORQHAT_API_KEY,
});

async function deleteRecords() {
  try {
    const response = await client.db.deleteRecords({
      table: 'users',
      conditions: {
        status: 'inactive',
        last_login_before: '2024-01-01T00:00:00Z'
      }
    });
    console.log(`Deleted ${response.deletedCount} records`);
    return response;
  } catch (error) {
    console.error('Error deleting records:', error);
  }
}

deleteRecords();
```

```python Python
from worqhat import Worqhat

# Initialize the client
client = Worqhat(
    api_key="YOUR_API_KEY",  # Better to use environment variables
)

# Delete records
response = client.db.delete_records(
    table="users",
    conditions={
        "status": "inactive",
        "last_login_before": "2024-01-01T00:00:00Z"
    }
)
print(f"Deleted {response.deleted_count} records")
```

```go Go
package main

import (
  "context"
  "fmt"

  "github.com/WorqHat/worqhat-go-sdk"
  "github.com/WorqHat/worqhat-go-sdk/option"
)

func main() {
  client := worqhat.NewClient(
    option.WithAPIKey("YOUR_API_KEY"),  // Better to use environment variables
  )
  
  response, err := client.DB.DeleteRecords(context.TODO(), worqhat.DBDeleteRecordsParams{
    Table: "users",
    Conditions: map[string]interface{}{
      "status": "inactive",
      "last_login_before": "2024-01-01T00:00:00Z",
    },
  })
  if err != nil {
    panic(err.Error())
  }
  
  fmt.Printf("Deleted %d records\n", response.DeletedCount)
}
```
